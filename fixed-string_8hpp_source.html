37/**
38 * Fixed size string class.
39 *
40 * fixed_string is a template class that encapsulates a string of a fixed number of characters.
41 * It is designed to be used where strings are needed as non-type template parameters.
42 *
43 * @tparam N The maximum number of characters the fixed_string can hold.
44 */
45template <size_t N>
46struct fixed_string {
47 using const_pointer = const char*; /**< Type of constant pointer to the string data. */
48 using const_reference = const char&; /**< Type of constant reference to a character in the string. */
49 using const_iterator = const_pointer; /**< Type of constant iterator for traversing the string. */
50 using const_reverse_iterator = std::reverse_iterator<const_iterator>; /**< Type of constant reverse iterator for traversing the string in reverse. */
51

54 size_t length = 0;

65 /**
66 * Constructs a fixed_string from a character array.
67 *
68 * This constructor template initializes a fixed_string with the contents of a given character array.
69 * Characters from encodings other than ASCII may be truncated.
70 *
71 * @tparam T The character type of the input array.
72 * @param[in] data The character array to initialize the fixed_string with.
73 */
74 template <typename T>
75 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE fixed_string(const T (&data)[N + 1]) IPADDRESS_NOEXCEPT {
76 internal::is_char_type<T>();
77 auto ended = false;
78 for (size_t i = 0; i < N; ++i) {
79 if (IPADDRESS_IS_CONST_EVALUATED(data) && data[i] > 127) {
80 const size_t err = data[i] / (data[i] - data[i]); // NOLINT(misc-redundant-expression): invalid symbol for ip address
81 }
82 _data[i] = char(data[i]);
83 if (data[i] == '\0') {
84 ended = true;
85 }
86 if (!ended) {
87 ++length;
88 }
89 }
90 }

308 /**
309 * Compares the string with another fixed_string.
310 *
311 * Compares the string with another fixed_string lexicographically.
312 *
313 * @tparam N2 The size of the other fixed_string.
314 * @param[in] rhs The other fixed_string to compare with.
315 * @return Negative value if less, zero if equal, positive value if greater.
316 */
317 template <size_t N2>
318 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE int compare(const fixed_string<N2>& rhs) const IPADDRESS_NOEXCEPT {
319 size_t i = 0;
320 for (; i < size() && i < rhs.size(); ++i) {
321 const auto c1 = at(i);
322 const auto c2 = rhs.at(i);
323 if (c1 != c2) {
324 return int(c1) - int(c2);
325 }
326 }
327
328 if (i == size() && i == rhs.size()) {
329 return 0;
330 } else if (i == size() && i < rhs.size()) {
331 return -1;
332 } else {
333 return 1;
334 }
335 }

337 /**
338 * Calculates the hash of the string.
339 *
340 * Computes a hash value for the string using a FNV-1a hash function.
341 *
342 * @return The hash value of the string.
343 */
344 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t hash() const IPADDRESS_NOEXCEPT {
345 size_t value{};
346 size_t prime{};
347
348 if (sizeof(size_t) == 8) {
349 value = size_t(14695981039346656037ULL);
350 prime = size_t(1099511628211ULL);
351 } else {
352 value = size_t(2166136261U);
353 prime = size_t(16777619U);
354 }
355
356 for (size_t i = 0; i < size(); ++i) {
357 value ^= static_cast<size_t>(_data[i]);
358 value *= prime;
359 }
360 return value;
361 }

363 /**
364 * Swaps the contents with another fixed_string.
365 *
366 * Exchanges the contents of the string with those of another fixed_string.
367 *
368 * @param[in,out] other The other fixed_string to swap with.
369 */
370 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void swap(fixed_string& other) IPADDRESS_NOEXCEPT {
371 auto count = size() < other.size() ? other.size() : size();
372 for (size_t i = 0; i < count; ++i) {
373 auto tmp = _data[i];
374 _data[i] = other._data[i];
375 other._data[i] = tmp;
376 }
377 auto tmp = length;
378 length = other.length;
379 other.length = tmp;
380 }
