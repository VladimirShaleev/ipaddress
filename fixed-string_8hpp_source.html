339 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t hash() const IPADDRESS_NOEXCEPT {
340 size_t value{};
341 size_t prime{};
342
343 if (sizeof(size_t) == 8) {
344 value = size_t(14695981039346656037ULL);
345 prime = size_t(1099511628211ULL);
346 } else {
347 value = size_t(2166136261U);
348 prime = size_t(16777619U);
349 }
350

352 value ^= static_cast<size_t>(_data[i]);
353 value *= prime;

366 auto count = size() < other.size() ? other.size() : size();
367 for (size_t i = 0; i < count; ++i) {
