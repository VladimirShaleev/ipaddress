952 uint32_t(quotient._lower),
953 uint32_t(quotient._lower >> 32),
954 uint32_t(quotient._upper),
955 uint32_t(quotient._upper >> 32)
956 };
957 int32_t left_size = 4 - int32_t(leading_zero_count(quotient)) / 32;
958
959 uint32_t right[4] {
960 uint32_t(divisor._lower),
961 uint32_t(divisor._lower >> 32),
962 uint32_t(divisor._upper),
963 uint32_t(divisor._upper >> 32)
964 };
965 int32_t right_size = 4 - int32_t(leading_zero_count(divisor)) / 32;
966
967 uint32_t bits[4]{};

969
970 assert(left_size >= 1);
971 assert(right_size >= 1);
972 assert(left_size >= right_size);
973
974 uint32_t div_hi = right[right_size - 1];
975 uint32_t div_lo = right_size > 1 ? right[right_size - 2] : 0;
976
977 int32_t shift = leading_zero_count(div_hi);
978 int32_t back_shift = 32 - shift;
979
980 if (shift > 0) {
981 uint32_t div_nx = right_size > 2 ? right[right_size - 3] : 0;
982 div_hi = (div_hi << shift) | (div_lo >> back_shift);
983 div_lo = (div_lo << shift) | (div_nx >> back_shift);
984 }

986 for (int32_t i = left_size; i >= right_size; --i) {
987 int32_t n = i - right_size;
988 uint32_t t = uint32_t(i) < uint32_t(left_size) ? left[i] : 0;
989
990 uint64_t val_hi = (uint64_t(t) << 32) | left[i - 1];
991 uint32_t val_lo = (i > 1) ? left[i - 2] : 0;
992
993 if (shift > 0) {
994 uint32_t val_nx = i > 2 ? left[i - 3] : 0;
995 val_hi = (val_hi << shift) | (val_lo >> back_shift);
996 val_lo = (val_lo << shift) | (val_nx >> back_shift);
997 }
998
999 uint64_t digit = val_hi / div_hi;
1000
1001 if (digit > 0xFFFFFFFF) {
1002 digit = 0xFFFFFFFF;
1003 }
1004
1005 while (divide_guess_too_big(digit, val_hi, val_lo, div_hi, div_lo)) {
1006 --digit;
1007 }

1011
1012 if (carry != t) {
1013 assert(carry == t + 1);
1014
1015 carry = add_divisor(left + n, left_size - n, right, right_size);
1016
1017 --digit;
1018 assert(carry == 1);

1023 bits[n] = uint32_t(digit);
