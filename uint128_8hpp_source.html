15 * optional type compatibility.
16 *
17 * The `uint128_t` type is especially useful in applications requiring precise control over
18 * large integer values, such as cryptography, high-precision arithmetic, and IP address manipulation.

1137 auto al = uint32_t(a);
1138 auto ah = uint32_t(a >> 32);
1139 auto bl = uint32_t(b);
1140 auto bh = uint32_t(b >> 32);
1141
1142 uint64_t mull = uint64_t(al) * bl;
1143 uint64_t t = uint64_t(ah) * bl + (mull >> 32);
1144 uint64_t tl = uint64_t(al) * bh + uint32_t(t);
1145
1146 lower = tl << 32 | uint32_t(mull);
1147
1148 return uint64_t(ah) * bh + (t >> 32) + (tl >> 32);
1149 }

1169 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint128_t divide_slow(const uint128_t& quotient, const uint128_t& divisor) IPADDRESS_NOEXCEPT {
1170 uint32_t left[4] {
1171 uint32_t(quotient._lower),
1172 uint32_t(quotient._lower >> 32),
1173 uint32_t(quotient._upper),
1174 uint32_t(quotient._upper >> 32)
1175 };
1176 int32_t left_size = 4 - int32_t(leading_zero_count(quotient)) / 32;
1177
1178 uint32_t right[4] {
1179 uint32_t(divisor._lower),
1180 uint32_t(divisor._lower >> 32),
1181 uint32_t(divisor._upper),
1182 uint32_t(divisor._upper >> 32)
1183 };
1184 int32_t right_size = 4 - int32_t(leading_zero_count(divisor)) / 32;
1185
1186 uint32_t bits[4]{};
1187 int32_t bits_size = left_size - right_size + 1;
1188
1189 assert(left_size >= 1);
1190 assert(right_size >= 1);
1191 assert(left_size >= right_size);
1192
1193 uint32_t div_hi = right[right_size - 1];
1194 uint32_t div_lo = right_size > 1 ? right[right_size - 2] : 0;
1195
1196 uint32_t shift = leading_zero_count(div_hi);
1197 uint32_t back_shift = 32 - shift;
1198
1199 if (shift > 0) {
1200 uint32_t div_nx = right_size > 2 ? right[right_size - 3] : 0;
1201 div_hi = (div_hi << shift) | (div_lo >> back_shift);
1202 div_lo = (div_lo << shift) | (div_nx >> back_shift);
1203 }
1204
1205 for (int32_t i = left_size; i >= right_size; --i) {
1206 int32_t n = i - right_size;
1207 uint32_t t = uint32_t(i) < uint32_t(left_size) ? left[i] : 0;
1208
1209 uint64_t val_hi = (uint64_t(t) << 32) | left[i - 1];
1210 uint32_t val_lo = (i > 1) ? left[i - 2] : 0; // NOLINT(clang-analyzer-core.uninitialized.Assign)
1211
1212 if (shift > 0) {
1213 uint32_t val_nx = i > 2 ? left[i - 3] : 0;
1214 val_hi = (val_hi << shift) | (val_lo >> back_shift);
1215 val_lo = (val_lo << shift) | (val_nx >> back_shift);
1216 }
1217
1218 uint64_t digit = val_hi / div_hi;
1219
1220 if (digit > 0xFFFFFFFF) {
1221 digit = 0xFFFFFFFF;
1222 }
1223
1224 while (divide_guess_too_big(digit, val_hi, val_lo, div_hi, div_lo)) {
1225 --digit;
1226 }
1227
1228 if (digit > 0) {
1229 uint32_t carry = subtract_divisor(left + n, left_size - n, right, right_size, digit);
1230
1231 if (carry != t) {
1232 assert(carry == t + 1);
1233
1234 // NOLINTNEXTLINE(clang-analyzer-deadcode.DeadStores): carry is not used in the release build
1235 carry = add_divisor(left + n, left_size - n, right, right_size);
1236
1237 --digit;
1238 assert(carry == 1);
1239 }
1240 }
1241
1242 if (uint32_t(n) < uint32_t(bits_size)) {
1243 bits[n] = uint32_t(digit);
1244 }
1245
1246 if (uint32_t(i) < uint32_t(left_size)) {
1247 left[i] = 0;
1248 }
1249 }
1250
1251 return { uint64_t(bits[3]) << 32 | bits[2], uint64_t(bits[1]) << 32 | bits[0] };
1252 }

1254 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool divide_guess_too_big(uint64_t q, uint64_t val_hi, uint32_t val_lo, uint32_t div_hi, uint32_t div_lo) IPADDRESS_NOEXCEPT {
1255 assert(q <= 0xFFFFFFFF);
1256
1257 uint64_t chk_hi = div_hi * q;
1258 uint64_t chk_lo = div_lo * q;
1259
1260 chk_hi += chk_lo >> 32;
1261 chk_lo = uint32_t(chk_lo);
1262
1263 return (chk_hi > val_hi) || (chk_hi == val_hi && chk_lo > val_lo);
1264 }

1266 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t subtract_divisor(uint32_t* left, int32_t ls /* NOLINT(misc-unused-parameters) */, const uint32_t (&right)[4], int32_t rs, uint64_t q) IPADDRESS_NOEXCEPT {
1267 assert(ls >= rs);
1268 assert(q <= 0xFFFFFFFF);
1269
1270 uint64_t carry = 0;
1271
1272 for (int32_t i = 0; i < rs; ++i) {
1273 carry += right[i] * q;
1274
1275 auto digit = uint32_t(carry);
1276 carry >>= 32;
1277
1278 if (left[i] < digit) {
1279 ++carry;
1280 }
1281 left[i] -= digit;
1282 }
1283
1284 return uint32_t(carry);
1285 }

1287 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t add_divisor(uint32_t* left, int32_t ls /* NOLINT(misc-unused-parameters) */, const uint32_t (&right)[4], int32_t rs) IPADDRESS_NOEXCEPT {
1288 assert(ls >= rs);
1289
1290 uint64_t carry = 0;
1291
1292 for (int32_t i = 0; i < rs; ++i) {
1293 uint64_t digit = left[i] + carry + right[i];
1294 left[i] = uint32_t(digit);
1295 carry = digit >> 32;
1296 }
1297
1298 return uint32_t(carry);
1299 }

1301 IPADDRESS_NODISCARD static IPADDRESS_FORCE_INLINE uint128_t from_double(double value) IPADDRESS_NOEXCEPT {
1302 constexpr double two_pow_128 = 340282366920938463463374607431768211456.0;
1303
1304 if (value < 0 || std::isnan(value)) {
1305 return uint128_t(0, 0);
1306 }
1307
1308 if (value >= two_pow_128) {
1309 return uint128_t(0xFFFFFFFFFFFFFFFF, 0xFFFFFFFFFFFFFFFF);
1310 }
1311
1312 if (value >= 1.0) {

1314 uint128_t result((bits << 12) >> 1 | 0x8000000000000000, 0x0000000000000000);
1315 result >>= (1023 + 128 - 1 - int32_t(bits >> 52));
1316 return result;
1317 }

1323 constexpr double two_pow_52 = 4503599627370496.0;
1324 constexpr double two_pow_76 = 75557863725914323419136.0;
1325 constexpr double two_pow_104 = 20282409603651670423947251286016.0;

1328 constexpr uint64_t two_pow_52_bits = 0x4330000000000000;
1329 constexpr uint64_t two_pow_76_bits = 0x44B0000000000000;
1330 constexpr uint64_t two_pow_104_bits = 0x4670000000000000;
1331 constexpr uint64_t two_pow_128_bits = 0x47F0000000000000;
1332
1333 if (value._upper == 0) {
1334 return double(value._lower);
1335 }
1336
1337 if ((value._upper >> 24) == 0) {
1338 double lower = uint64_bits_to_double(two_pow_52_bits | ((value._lower << 12) >> 12)) - two_pow_52;

1344 double upper = uint64_bits_to_double(two_pow_128_bits | (uint64_t)(value >> 76)) - two_pow_128;
1345 return double(lower + upper);
