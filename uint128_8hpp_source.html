994  uint32_t val_nx = i > 2 ? left[i - 3] : 0;
995  val_hi = (val_hi << shift) | (val_lo >> back_shift);
996  val_lo = (val_lo << shift) | (val_nx >> back_shift);
997  }
998 
999  uint64_t digit = val_hi / div_hi;
1000 
1001  if (digit > 0xFFFFFFFF) {
1002  digit = 0xFFFFFFFF;
1003  }
1004 
1005  while (divide_guess_too_big(digit, val_hi, val_lo, div_hi, div_lo)) {
1006  --digit;
1007  }
1008 
1009  if (digit > 0) {
1010  uint32_t carry = subtract_divisor(left + n, left_size - n, right, right_size, digit);
1011 
1012  if (carry != t) {
1013  assert(carry == t + 1);
1014 
1015  carry = add_divisor(left + n, left_size - n, right, right_size);
1016 
1017  --digit;
1018  assert(carry == 1);
1019  }
1020  }
1021 
1022  if (uint32_t(n) < uint32_t(bits_size)) {
1023  bits[n] = uint32_t(digit);
1024  }
1025 
1026  if (uint32_t(i) < uint32_t(left_size)) {
1027  left[i] = 0;
1028  }
1029  }
1030 
1031  return { uint64_t(bits[3]) << 32 | bits[2], uint64_t(bits[1]) << 32 | bits[0] };
1032  }

1034  IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool divide_guess_too_big(uint64_t q, uint64_t val_hi, uint32_t val_lo, uint32_t div_hi, uint32_t div_lo) IPADDRESS_NOEXCEPT {
1035  assert(q <= 0xFFFFFFFF);
1036 
1037  uint64_t chk_hi = div_hi * q;
1038  uint64_t chk_lo = div_lo * q;
1039 
1040  chk_hi += chk_lo >> 32;
1041  chk_lo = uint32_t(chk_lo);
1042 
1043  return (chk_hi > val_hi) || (chk_hi == val_hi && chk_lo > val_lo);
1044  }

1046  IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t subtract_divisor(uint32_t* left, size_t ls, uint32_t (&right)[4], size_t rs, uint64_t q) IPADDRESS_NOEXCEPT {
1047  assert(ls >= rs);
1048  assert(q <= 0xFFFFFFFF);
1049 
1050  uint64_t carry = 0;
1051 
1052  for (int32_t i = 0; i < rs; ++i) {
1053  carry += right[i] * q;
1054 
1055  uint32_t digit = uint32_t(carry);
1056  carry >>= 32;
1057 
1058  uint32_t& left_item = left[i];
1059 
1060  if (left_item < digit) {
1061  ++carry;
1062  }
1063  left_item -= digit;
1064  }
1065 
1066  return uint32_t(carry);
1067  }

1070  assert(ls >= rs);
1071 
1072  uint64_t carry = 0;
1073 
1074  for (int32_t i = 0; i < rs; ++i) {
1075  uint32_t& left_element = left[i];
1076  uint64_t digit = left_element + carry + right[i];
1077 
1078  left_element = uint32_t(digit);
1079  carry = digit >> 32;
1080  }
