1175 uint32_t al = uint32_t(a);
1176 uint32_t ah = uint32_t(a >> 32);
1177 uint32_t bl = uint32_t(b);
1178 uint32_t bh = uint32_t(b >> 32);
1179

1181 uint64_t t = uint64_t(ah) * bl + (mull >> 32);
1182 uint64_t tl = uint64_t(al) * bh + uint32_t(t);
1183
1184 lower = tl << 32 | uint32_t(mull);
1185
1186 return uint64_t(ah) * bh + (t >> 32) + (tl >> 32);
1187 }

1207 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint128_t divide_slow(const uint128_t& quotient, const uint128_t& divisor) IPADDRESS_NOEXCEPT {
1208 uint32_t left[4] {
1209 uint32_t(quotient._lower),
1210 uint32_t(quotient._lower >> 32),
1211 uint32_t(quotient._upper),
1212 uint32_t(quotient._upper >> 32)
1213 };
1214 int32_t left_size = 4 - int32_t(leading_zero_count(quotient)) / 32;
1215
1216 uint32_t right[4] {
1217 uint32_t(divisor._lower),
1218 uint32_t(divisor._lower >> 32),
1219 uint32_t(divisor._upper),
1220 uint32_t(divisor._upper >> 32)
1221 };
1222 int32_t right_size = 4 - int32_t(leading_zero_count(divisor)) / 32;
1223
1224 uint32_t bits[4]{};
1225 int32_t bits_size = left_size - right_size + 1;
1226
1227 assert(left_size >= 1);
1228 assert(right_size >= 1);
1229 assert(left_size >= right_size);
1230
1231 uint32_t div_hi = right[right_size - 1];
1232 uint32_t div_lo = right_size > 1 ? right[right_size - 2] : 0;
1233
1234 int32_t shift = leading_zero_count(div_hi);
1235 int32_t back_shift = 32 - shift;
1236
1237 if (shift > 0) {
1238 uint32_t div_nx = right_size > 2 ? right[right_size - 3] : 0;
1239 div_hi = (div_hi << shift) | (div_lo >> back_shift);
1240 div_lo = (div_lo << shift) | (div_nx >> back_shift);
1241 }
1242
1243 for (int32_t i = left_size; i >= right_size; --i) {
1244 int32_t n = i - right_size;
1245 uint32_t t = uint32_t(i) < uint32_t(left_size) ? left[i] : 0;
1246
1247 uint64_t val_hi = (uint64_t(t) << 32) | left[i - 1];
1248 uint32_t val_lo = (i > 1) ? left[i - 2] : 0;
1249
1250 if (shift > 0) {
1251 uint32_t val_nx = i > 2 ? left[i - 3] : 0;
1252 val_hi = (val_hi << shift) | (val_lo >> back_shift);
1253 val_lo = (val_lo << shift) | (val_nx >> back_shift);
1254 }
1255
1256 uint64_t digit = val_hi / div_hi;
1257
1258 if (digit > 0xFFFFFFFF) {
1259 digit = 0xFFFFFFFF;
1260 }
1261
1262 while (divide_guess_too_big(digit, val_hi, val_lo, div_hi, div_lo)) {
1263 --digit;
1264 }
1265
1266 if (digit > 0) {
1267 uint32_t carry = subtract_divisor(left + n, left_size - n, right, right_size, digit);
1268
1269 if (carry != t) {
1270 assert(carry == t + 1);
1271
1272 carry = add_divisor(left + n, left_size - n, right, right_size);
1273
1274 --digit;
1275 assert(carry == 1);
1276 }
1277 }
1278
1279 if (uint32_t(n) < uint32_t(bits_size)) {
1280 bits[n] = uint32_t(digit);
1281 }
1282
1283 if (uint32_t(i) < uint32_t(left_size)) {
1284 left[i] = 0;
1285 }
1286 }
1287
1288 return { uint64_t(bits[3]) << 32 | bits[2], uint64_t(bits[1]) << 32 | bits[0] };
1289 }

1291 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool divide_guess_too_big(uint64_t q, uint64_t val_hi, uint32_t val_lo, uint32_t div_hi, uint32_t div_lo) IPADDRESS_NOEXCEPT {
1292 assert(q <= 0xFFFFFFFF);
1293
1294 uint64_t chk_hi = div_hi * q;
1295 uint64_t chk_lo = div_lo * q;
1296
1297 chk_hi += chk_lo >> 32;
1298 chk_lo = uint32_t(chk_lo);
1299
1300 return (chk_hi > val_hi) || (chk_hi == val_hi && chk_lo > val_lo);
1301 }

1303 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t subtract_divisor(uint32_t* left, size_t ls, uint32_t (&right)[4], size_t rs, uint64_t q) IPADDRESS_NOEXCEPT {
1304 assert(ls >= rs);
1305 assert(q <= 0xFFFFFFFF);
1306
1307 uint64_t carry = 0;
1308
1309 for (int32_t i = 0; i < rs; ++i) {
1310 carry += right[i] * q;
1311
1312 uint32_t digit = uint32_t(carry);
1313 carry >>= 32;
1314
1315 uint32_t& left_item = left[i];
1316
1317 if (left_item < digit) {
1318 ++carry;
1319 }
1320 left_item -= digit;
1321 }
1322
1323 return uint32_t(carry);
1324 }

1326 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t add_divisor(uint32_t* left, size_t ls, uint32_t (&right)[4], size_t rs) IPADDRESS_NOEXCEPT {
1327 assert(ls >= rs);
1328
1329 uint64_t carry = 0;
1330
1331 for (int32_t i = 0; i < rs; ++i) {
1332 uint32_t& left_element = left[i];
1333 uint64_t digit = left_element + carry + right[i];
1334
1335 left_element = uint32_t(digit);
1336 carry = digit >> 32;
1337 }
1338
1339 return uint32_t(carry);
1340 }

1343 constexpr double two_pow_128 = 340282366920938463463374607431768211456.0;
1344
1345 if (value < 0 || std::isnan(value)) {
1346 return uint128_t(0, 0);
