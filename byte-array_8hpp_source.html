6 *
7 * The byte_array template class is designed to encapsulate a fixed-size array of bytes (uint8_t),
8 * providing various type definitions for ease of use, and methods for accessing the array elements.
9 * Intended for use in managing IP address representations and manipulations. The class ensures that
10 * the array size is constant and known at compile-time, offering a compile-time guarantee of array
11 * bounds, which can prevent common errors associated with dynamic arrays.

21/**
22 * A template class for creating and managing a fixed-size array of bytes.
23 *
24 * The byte_array class encapsulates a static-size array of bytes, providing type definitions
25 * for element access and iteration in both normal and reverse order. It is designed to be used
26 * where a constant size byte buffer is needed, such as in handling network addresses like IP addresses.
27 *
28 * @tparam N The number of bytes in the array.
29 * @remark The purpose of the byte_array class is to provide functionality similar to std::array in .
30 * environments where std::array cannot be used to its full extent during compile-time operations.
31 */

422 /**
423 * Compares two byte_array objects for ordering.
424 *
425 * Performs a lexicographical comparison between two byte_array objects using the three-way comparison operator.
426 * The comparison is done element-wise and stops at the first unequal pair of elements, returning the result
427 * of comparing these two elements. If all elements are equal, the byte_arrays are considered equivalent.
428 *
429 * @tparam N The size of the byte array.
430 * @param[in] lhs A reference to the left-hand side byte_array object.
431 * @param[in] rhs A reference to the right-hand side byte_array object.
432 * @return An `std::strong_ordering` value indicating the ordering relationship.
433 * @retval std::strong_ordering::less if lhs is lexicographically less than rhs
434 * @retval std::strong_ordering::greater if lhs is lexicographically greater than rhs
435 * @retval std::strong_ordering::equivalent if lhs is lexicographically equal to rhs
436 */
437 template <std::size_t N>
438 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE std::strong_ordering operator<=>(const byte_array<N>& lhs, const byte_array<N>& rhs) IPADDRESS_NOEXCEPT {
439 for (std::size_t i = 0; i < N; ++i) {
440 if (const auto result = lhs[i] <=> rhs[i]; result != std::strong_ordering::equivalent) {
441 return result;
442 }
443 }
444 return std::strong_ordering::equivalent;
445 }
