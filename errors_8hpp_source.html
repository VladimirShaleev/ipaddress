1/**
2 * @file errors.hpp
3 * @brief Declares error codes and exceptions for IP address operations
4 * @author Vladimir Shaleev
5 * @copyright MIT License
6 *
7 * This header file is part of an IP address processing library and is responsible
8 * for defining error codes and exceptions that represent various failure states
9 * encountered during IP address parsing and processing. It includes an enumeration
10 * of error codes that cover a range of issues from syntactical errors in IP address
11 * strings to logical inconsistencies in network operations. Additionally, the file
12 * declares a base exception class tailored to encapsulate these error codes along
13 * with descriptive messages, facilitating precise error handling and reporting.
14 *
15 * The error codes are categorized into general parsing errors, IPv4-specific errors,
16 * IPv6-specific errors, and logical errors, each with a brief description of the
17 * condition they represent. The base exception class, `error`, inherits from the
18 * standard `std::runtime_error` and is constructed with an error code and a message,
19 * allowing exceptions to carry both a machine-readable identifier and a human-readable
20 * explanation.
21 *
22 * The file also includes a function template that converts an error code into an
23 * exception, streamlining the process of throwing exceptions upon encountering
24 * errors during IP address operations. This design ensures that all potential error
25 * conditions are accounted for and can be handled gracefully by client code using
26 * the library.
27 */

36/**
37 * Enumeration of error codes for IP address parsing and validation.
38 *
39 * This enumeration defines a set of error codes used to represent various error conditions
40 * that can occur during the parsing and validation of IP addresses. These error codes are
41 * used to provide detailed feedback about the nature of the error encountered, allowing
42 * for more precise error handling and troubleshooting.
43 *
44 * The error codes are divided into several categories:
45 * - General errors related to the overall format and structure of the IP address.
46 * - IPv4-specific errors that address the unique syntactical requirements of IPv4 addresses.
47 * - IPv6-specific errors that address the unique syntactical requirements of IPv6 addresses.
48 * - Logical errors that represent inconsistencies or invalid operations in network calculations.
49 *
50 * Each error code is accompanied by a brief comment explaining the specific error condition it represents.
51 */
52enum class error_code {
53 NO_ERROR = 0, /**< Indicates the absence of any errors. */
54 EMPTY_ADDRESS, /**< The IP address string is empty when it should contain a valid address. */
55 EMPTY_NETMASK, /**< The netmask portion of the address is empty when it should specify a valid netmask. */
56 INVALID_NETMASK, /**< The provided netmask is not valid according to standard netmask formatting rules. */
57 NETMASK_PATTERN_MIXES_ZEROES_AND_ONES, /**< The netmask contains an invalid pattern of zeroes and ones. */
58 HAS_HOST_BITS_SET, /**< The address has host bits set when they are expected to be clear. */
59 ONLY_ONE_SLASH_PERMITTED, /**< Only one slash character is permitted, used to separate the address from the netmask. */
60
61 // ipv4 errors
62 EMPTY_OCTET, /**< An octet in the IPv4 address is empty when it should contain a numeric value. */
63 EXPECTED_4_OCTETS, /**< The IPv4 address does not contain the expected four octets. */
64 LEADING_0_ARE_NOT_PERMITTED, /**< Leading zeroes are not permitted in any octet of the IPv4 address. */
65 OCTET_MORE_3_CHARACTERS, /**< An octet contains more than three characters, exceeding the maximum allowed. */
66 OCTET_HAS_INVALID_SYMBOL, /**< An octet contains characters other than digits, which are invalid. */
67 OCTET_EXCEEDED_255, /**< An octet's value exceeds the maximum allowed value of 255. */
68
69 // ipv6 errors
70 LEAST_3_PARTS, /**< The IPv6 address contains fewer than the minimum required parts. */
71 MOST_8_COLONS_PERMITTED, /**< The IPv6 address contains more than the maximum allowed number of colons. */
72 PART_IS_MORE_4_CHARS, /**< A part of the IPv6 address contains more than four characters. */
73 PART_HAS_INVALID_SYMBOL, /**< A part of the IPv6 address contains invalid characters. */
74 MOST_ONE_DOUBLE_COLON_PERMITTED, /**< More than one double colon is present in the IPv6 address. */
75 LEADING_COLON_ONLY_PERMITTED_AS_PART_OF_DOUBLE_COLON, /**< A leading colon is only permitted as part of a double colon. */
76 TRAILING_COLON_ONLY_PERMITTED_AS_PART_OF_DOUBLE_COLON, /**< A trailing colon is only permitted as part of a double colon. */

84 INVALID_PREFIXLEN_DIFF, /**< The difference in prefix length is invalid for the operation being performed. */
85 NEW_PREFIX_MUST_BE_SHORTER, /**< The new prefix length must be shorter for the operation being performed. */

91/**
92 * The primary exception class used by the IP address library.
93 *
94 * This class extends the standard `std::runtime_error` and serves as the base exception
95 * type for the library. It is designed to be thrown when an error condition is encountered
96 * during the processing of IP addresses, such as parsing errors or logical inconsistencies.
97 *
98 * The `error` class encapsulates an `error_code` value representing the specific error
99 * that occurred, along with a descriptive message that provides context about the error.
100 * This design allows for precise and informative error reporting, aiding in debugging
101 * and error handling.
102 *
103 * Overall, the `error` class is a critical component of the library's error management
104 * system, providing a standardized way to handle exceptions throughout the library.
105 *
106 * @sa parse_error
107 * @sa logic_error
108 */
109class error : public std::runtime_error {
110public:
111 /**
112 * Constructs an error with a code and a concatenated message from multiple arguments.
113 *
114 * @tparam FirstArg The type of the first argument passed to the constructor.
115 * @tparam Args The types of the additional arguments passed to the constructor.
116 * @param[in] code The error code associated with the exception.
117 * @param[in] arg The first argument describing the error.
118 * @param[in] args Additional arguments describing the error.
119 */
120 template <typename FirstArg, typename... Args>
121 explicit error(error_code code, const FirstArg& arg, const Args&... args) : std::runtime_error(concatenate(arg, args...)), _code(code) {
122 }
123
124 /**
125 * Constructs an error with a code and a message string.
126 *
127 * @param[in] code The error code associated with the exception.
128 * @param[in] message A detailed description of the error.
129 */
130 explicit error(error_code code, const std::string& message) : std::runtime_error(message), _code(code) {
131 }
132
133 /**
134 * Constructs an error with a code and a message C-string.
135 *
136 * @param[in] code The error code associated with the exception.
137 * @param[in] message A detailed description of the error.
138 */
139 explicit error(error_code code, const char* message) : std::runtime_error(message), _code(code) {
140 }
141
142 /**
143 * Returns the error code associated with this error.
144 *
145 * @return The error code that represents the specific error condition.
146 */
147 IPADDRESS_NODISCARD error_code code() const IPADDRESS_NOEXCEPT {
148 return _code;
149 }
150
151private:
152 template <typename... Args>
153 static std::string concatenate(const Args&... args) {
154 std::ostringstream ss;
155 print(ss, args...);
156 return ss.str();
157 }
158
159 template <typename FirstArg, typename... Args>
160 static void print(std::ostringstream& out, const FirstArg& arg, const Args&... args) {
161 out << arg << ' ';
162 print(out, args...);
163 }
164
165 template <typename FirstArg>
166 static void print(std::ostringstream& out, const FirstArg& arg) {
167 out << arg;
168 }
169
170 error_code _code;
171};

261/**
262 * Raises an error with a specific error code and additional context.
263 *
264 * This function constructs an error message based on the provided error code,
265 * index, and address, then throws a parse_error or a logic_error exception with the constructed message.
266 *
267 * @tparam T The character type of the address string.
268 * @param[in] code The error code indicating the type of error encountered.
269 * @param[in] index The index at which the error occurred, if applicable.
270 * @param[in] address A pointer to the beginning of the address string.
271 * @param[in] length The length of the address string.
272 * @throw parse_error Thrown with a message corresponding to the error code.
273 * @throw logic_error Thrown with a message corresponding to the error code.
274 * @note This function is marked [[noreturn]] as it always throws an exception.
275 */
276template <typename T>
277[[noreturn]] IPADDRESS_CONSTEXPR inline void raise_error(error_code code, int index, const T* address, size_t length) {
278 char str[101] = {};
279 size_t max_len = length;
280 if (length > 100) {
281 max_len = 100;
282 }
283 for (size_t i = 0; i < max_len; ++i) {
284 str[i] = char(address[i]);
285 }
286 switch (code) {
287 case error_code::EMPTY_ADDRESS:
288 throw parse_error(code, "address cannot be empty");
289 case error_code::EMPTY_NETMASK:
290 throw parse_error(code, "empty mask in address", str);
291 case error_code::INVALID_NETMASK:
292 throw parse_error(code, "is not a valid netmask in address", str);
293 case error_code::NETMASK_PATTERN_MIXES_ZEROES_AND_ONES:
294 throw parse_error(code, "netmask pattern mixes zeroes & ones in address", str);
295 case error_code::HAS_HOST_BITS_SET:
296 throw parse_error(code, "has host bits set in address", str);
297 case error_code::ONLY_ONE_SLASH_PERMITTED:
298 throw parse_error(code, "only one '/' permitted in address", str);
299 case error_code::EMPTY_OCTET:
300 throw parse_error(code, "empty octet", index, "in address", str);
301 case error_code::EXPECTED_4_OCTETS:
302 throw parse_error(code, "expected 4 octets in", str);
303 case error_code::LEADING_0_ARE_NOT_PERMITTED:
304 throw parse_error(code, "leading zeros are not permitted in octet", index, "of address", str);
305 case error_code::OCTET_MORE_3_CHARACTERS:
306 throw parse_error(code, "in octet", index, "of address", str, "more 3 characters");
307 case error_code::OCTET_HAS_INVALID_SYMBOL:
308 throw parse_error(code, "in octet", index, "of address", str, "has invalid symbol");
309 case error_code::OCTET_EXCEEDED_255:
310 throw parse_error(code, "octet", index, "of address", str, "exceeded 255");
311 case error_code::LEAST_3_PARTS:
312 throw parse_error(code, "least 3 parts in address", str);
313 case error_code::MOST_8_COLONS_PERMITTED:
314 throw parse_error(code, "most 8 colons permitted in address", str);
315 case error_code::PART_IS_MORE_4_CHARS:
316 throw parse_error(code, "in part", index, "of address", str, "more 4 characters");
317 case error_code::PART_HAS_INVALID_SYMBOL:
318 throw parse_error(code, "in part", index, "of address", str, "has invalid symbols");
319 case error_code::MOST_ONE_DOUBLE_COLON_PERMITTED:
320 throw parse_error(code, "at most one '::' permitted in address", str);
321 case error_code::LEADING_COLON_ONLY_PERMITTED_AS_PART_OF_DOUBLE_COLON:
322 throw parse_error(code, "at leading ':' only permitted as part of '::' in address", str);
323 case error_code::TRAILING_COLON_ONLY_PERMITTED_AS_PART_OF_DOUBLE_COLON:
324 throw parse_error(code, "at trailing ':' only permitted as part of '::' in address", str);
325 case error_code::EXPECTED_AT_MOST_7_OTHER_PARTS_WITH_DOUBLE_COLON:
326 throw parse_error(code, "expected at most 7 other parts with '::' in address", str);
327 case error_code::EXACTLY_8_PARTS_EXPECTED_WITHOUT_DOUBLE_COLON:
328 throw parse_error(code, "exactly 8 parts expected without '::' in address", str);
329 case error_code::SCOPE_ID_IS_TOO_LONG:
330 throw parse_error(code, "scope id is too long in address", str);
331 case error_code::INVALID_SCOPE_ID:
332 throw parse_error(code, "invalid scope id in address", str);
333 case error_code::INVALID_VERSION:
334 throw logic_error(code, "versions don't match");
335 case error_code::INVALID_PREFIXLEN_DIFF:
336 throw logic_error(code, "invalid prefixlen_diff");
337 case error_code::NEW_PREFIX_MUST_BE_SHORTER:
338 throw logic_error(code, "new prefix must be shorter");
339 case error_code::NEW_PREFIX_MUST_BE_LONGER:
340 throw logic_error(code, "new prefix must be longer");
341 case error_code::CANNOT_SET_PREFIXLEN_DIFF_AND_NEW_PREFIX:
342 throw logic_error(code, "cannot set prefixlen_diff and new_prefix");
343 case error_code::NOT_CONTAINED_NETWORK:
344 throw logic_error(code, "network is not a subnet of other");
345 default:
346 throw error(code, "unknown error");
347 }
