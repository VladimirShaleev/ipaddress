1/**
2 * @file errors.hpp
3 * @brief Declares error codes and exceptions for IP address operations
4 * @author Vladimir Shaleev
5 * @copyright MIT License
6 *
7 * This header file is part of an IP address processing library and is responsible
8 * for defining error codes and exceptions that represent various failure states
9 * encountered during IP address parsing and processing. It includes an enumeration
10 * of error codes that cover a range of issues from syntactical errors in IP address
11 * strings to logical inconsistencies in network operations. Additionally, the file
12 * declares a base exception class tailored to encapsulate these error codes along
13 * with descriptive messages, facilitating precise error handling and reporting.
14 *
15 * The error codes are categorized into general parsing errors, IPv4-specific errors,
16 * IPv6-specific errors, and logical errors, each with a brief description of the
17 * condition they represent. The base exception class, `error`, inherits from the
18 * standard `std::runtime_error` and is constructed with an error code and a message,
19 * allowing exceptions to carry both a machine-readable identifier and a human-readable
20 * explanation.
21 *
22 * The file also includes a function template that converts an error code into an
23 * exception, streamlining the process of throwing exceptions upon encountering
24 * errors during IP address operations. This design ensures that all potential error
25 * conditions are accounted for and can be handled gracefully by client code using
26 * the library.
27 */

36/**
37 * Enumeration of error codes for IP address parsing and validation.
38 *
39 * This enumeration defines a set of error codes used to represent various error conditions
40 * that can occur during the parsing and validation of IP addresses. These error codes are
41 * used to provide detailed feedback about the nature of the error encountered, allowing
42 * for more precise error handling and troubleshooting.
43 *
44 * The error codes are divided into several categories:
45 * - General errors related to the overall format and structure of the IP address.
46 * - IPv4-specific errors that address the unique syntactical requirements of IPv4 addresses.
47 * - IPv6-specific errors that address the unique syntactical requirements of IPv6 addresses.
48 * - Logical errors that represent inconsistencies or invalid operations in network calculations.
49 *
50 * Each error code is accompanied by a brief comment explaining the specific error condition it represents.
51 */
52enum class error_code {
53 no_error = 0, /**< Indicates the absence of any errors. */
54 empty_address, /**< The IP address string is empty when it should contain a valid address. */
55 empty_netmask, /**< The netmask portion of the address is empty when it should specify a valid netmask. */
56 invalid_netmask, /**< The provided netmask is not valid according to standard netmask formatting rules. */
57 netmask_pattern_mixes_zeroes_and_ones, /**< The netmask contains an invalid pattern of zeroes and ones. */
58 has_host_bits_set, /**< The address has host bits set when they are expected to be clear. */
59 only_one_slash_permitted, /**< Only one slash character is permitted, used to separate the address from the netmask. */
60 string_is_too_long, /**< Input string is too long. */
61
62 // ipv4 errors
63 empty_octet, /**< An octet in the IPv4 address is empty when it should contain a numeric value. */
64 expected_4_octets, /**< The IPv4 address does not contain the expected four octets. */
65 leading_0_are_not_permitted, /**< Leading zeroes are not permitted in any octet of the IPv4 address. */
66 octet_more_3_characters, /**< An octet contains more than three characters, exceeding the maximum allowed. */
67 octet_has_invalid_symbol, /**< An octet contains characters other than digits, which are invalid. */
68 octet_exceeded_255, /**< An octet's value exceeds the maximum allowed value of 255. */
69
70 // ipv6 errors
71 least_3_parts, /**< The IPv6 address contains fewer than the minimum required parts. */
72 most_8_colons_permitted, /**< The IPv6 address contains more than the maximum allowed number of colons. */
73 part_is_more_4_chars, /**< A part of the IPv6 address contains more than four characters. */
74 part_has_invalid_symbol, /**< A part of the IPv6 address contains invalid characters. */
75 most_one_double_colon_permitted, /**< More than one double colon is present in the IPv6 address. */
76 leading_colon_only_permitted_as_part_of_double_colon, /**< A leading colon is only permitted as part of a double colon. */
77 trailing_colon_only_permitted_as_part_of_double_colon, /**< A trailing colon is only permitted as part of a double colon. */
78 expected_at_most_7_other_parts_with_double_colon, /**< With a double colon present, at most seven other parts are expected. */
79 exactly_8_parts_expected_without_double_colon, /**< Without a double colon, exactly eight parts are expected. */
80 scope_id_is_too_long, /**< The scope ID in the IPv6 address exceeds the maximum length. */
81 invalid_scope_id, /**< The scope ID in the IPv6 address is invalid. */
82
83 // logic errors
84 invalid_version, /**< The IP address version does not match the expected version. */
85 invalid_prefixlen_diff, /**< The difference in prefix length is invalid for the operation being performed. */
86 new_prefix_must_be_shorter, /**< The new prefix length must be shorter for the operation being performed. */
87 new_prefix_must_be_longer, /**< The new prefix length must be longer for the operation being performed. */
88 cannot_set_prefixlen_diff_and_new_prefix, /**< Both prefix length difference and new prefix cannot be set simultaneously. */
89 not_contained_network /**< The network is not a subnet of the other network as expected. */
90};

92/**
93 * The primary exception class used by the IP address library.
94 *
95 * This class extends the standard `std::runtime_error` and serves as the base exception
96 * type for the library. It is designed to be thrown when an error condition is encountered
97 * during the processing of IP addresses, such as parsing errors or logical inconsistencies.
98 *
99 * The `error` class encapsulates an `error_code` value representing the specific error
100 * that occurred, along with a descriptive message that provides context about the error.
101 * This design allows for precise and informative error reporting, aiding in debugging
102 * and error handling.
103 *
104 * Overall, the `error` class is a critical component of the library's error management
105 * system, providing a standardized way to handle exceptions throughout the library.
106 *
107 * @sa parse_error
108 * @sa logic_error
109 */
110class error : public std::runtime_error {
111public:
112 /**
113 * Constructs an error with a code and a concatenated message from multiple arguments.
114 *
115 * @tparam FirstArg The type of the first argument passed to the constructor.
116 * @tparam Args The types of the additional arguments passed to the constructor.
117 * @param[in] code The error code associated with the exception.
118 * @param[in] arg The first argument describing the error.
119 * @param[in] args Additional arguments describing the error.
120 */
121 template <typename FirstArg, typename... Args>
122 explicit error(error_code code, const FirstArg& arg, const Args&... args) : std::runtime_error(concatenate(arg, args...)), _code(code) {
123 }
124
125 /**
126 * Constructs an error with a code and a message string.
127 *
128 * @param[in] code The error code associated with the exception.
129 * @param[in] message A detailed description of the error.
130 */
131 explicit error(error_code code, const std::string& message) : std::runtime_error(message), _code(code) {
132 }
133
134 /**
135 * Constructs an error with a code and a message C-string.
136 *
137 * @param[in] code The error code associated with the exception.
138 * @param[in] message A detailed description of the error.
139 */
140 explicit error(error_code code, const char* message) : std::runtime_error(message), _code(code) {
141 }
142
143 /**
144 * Returns the error code associated with this error.
145 *
146 * @return The error code that represents the specific error condition.
147 */
148 IPADDRESS_NODISCARD error_code code() const IPADDRESS_NOEXCEPT {
149 return _code;
150 }
151
152private:
153 template <typename... Args>
154 static std::string concatenate(const Args&... args) {
155 std::ostringstream ss;
156 print(ss, args...);
157 return ss.str();
158 }
159
160 template <typename FirstArg, typename... Args>
161 static void print(std::ostringstream& out, const FirstArg& arg, const Args&... args) {
162 out << arg << ' ';
163 print(out, args...);
164 }
165
166 template <typename FirstArg>
167 static void print(std::ostringstream& out, const FirstArg& arg) {
168 out << arg;
169 }
170
171 error_code _code;
172};

262/**
263 * Raises an error with a specific error code and additional context.
264 *
265 * This function constructs an error message based on the provided error code,
266 * index, and address, then throws a parse_error or a logic_error exception with the constructed message.
267 *
268 * @tparam T The character type of the address string.
269 * @param[in] code The error code indicating the type of error encountered.
270 * @param[in] index The index at which the error occurred, if applicable.
271 * @param[in] address A pointer to the beginning of the address string.
272 * @param[in] length The length of the address string.
273 * @throw parse_error Thrown with a message corresponding to the error code.
274 * @throw logic_error Thrown with a message corresponding to the error code.
275 * @note This function is marked [[noreturn]] as it always throws an exception.
276 */
277template <typename T>
278#ifndef IPADDRESS_NO_EXCEPTIONS
279[[noreturn]]
280#endif
281IPADDRESS_CONSTEXPR inline void raise_error(error_code code, int index, const T* address, size_t length) {
282#ifndef IPADDRESS_NO_EXCEPTIONS
283 char str[101] = {};
284 size_t max_len = length;
285 if (length > 100) {
286 max_len = 100;
287 }
288 for (size_t i = 0; i < max_len; ++i) {
289 str[i] = char(address[i]);
290 }
291 switch (code) {
292 case error_code::empty_address:
293 throw parse_error(code, "address cannot be empty");
294 case error_code::empty_netmask:
295 throw parse_error(code, "empty mask in address", str);
296 case error_code::invalid_netmask:
297 throw parse_error(code, "is not a valid netmask in address", str);
298 case error_code::netmask_pattern_mixes_zeroes_and_ones:
299 throw parse_error(code, "netmask pattern mixes zeroes & ones in address", str);
300 case error_code::has_host_bits_set:
301 throw parse_error(code, "has host bits set in address", str);
302 case error_code::only_one_slash_permitted:
303 throw parse_error(code, "only one '/' permitted in address", str);
304 case error_code::string_is_too_long:
305 throw parse_error(code, "input string is too long", str);
306 case error_code::empty_octet:
307 throw parse_error(code, "empty octet", index, "in address", str);
308 case error_code::expected_4_octets:
309 throw parse_error(code, "expected 4 octets in", str);
310 case error_code::leading_0_are_not_permitted:
311 throw parse_error(code, "leading zeros are not permitted in octet", index, "of address", str);
312 case error_code::octet_more_3_characters:
313 throw parse_error(code, "in octet", index, "of address", str, "more 3 characters");
314 case error_code::octet_has_invalid_symbol:
315 throw parse_error(code, "in octet", index, "of address", str, "has invalid symbol");
316 case error_code::octet_exceeded_255:
317 throw parse_error(code, "octet", index, "of address", str, "exceeded 255");
318 case error_code::least_3_parts:
319 throw parse_error(code, "least 3 parts in address", str);
320 case error_code::most_8_colons_permitted:
321 throw parse_error(code, "most 8 colons permitted in address", str);
322 case error_code::part_is_more_4_chars:
323 throw parse_error(code, "in part", index, "of address", str, "more 4 characters");
324 case error_code::part_has_invalid_symbol:
325 throw parse_error(code, "in part", index, "of address", str, "has invalid symbols");
326 case error_code::most_one_double_colon_permitted:
327 throw parse_error(code, "at most one '::' permitted in address", str);
328 case error_code::leading_colon_only_permitted_as_part_of_double_colon:
329 throw parse_error(code, "at leading ':' only permitted as part of '::' in address", str);
330 case error_code::trailing_colon_only_permitted_as_part_of_double_colon:
331 throw parse_error(code, "at trailing ':' only permitted as part of '::' in address", str);
332 case error_code::expected_at_most_7_other_parts_with_double_colon:
333 throw parse_error(code, "expected at most 7 other parts with '::' in address", str);
334 case error_code::exactly_8_parts_expected_without_double_colon:
335 throw parse_error(code, "exactly 8 parts expected without '::' in address", str);
336 case error_code::scope_id_is_too_long:
337 throw parse_error(code, "scope id is too long in address", str);
338 case error_code::invalid_scope_id:
339 throw parse_error(code, "invalid scope id in address", str);
340 case error_code::invalid_version:
341 throw logic_error(code, "versions don't match");
342 case error_code::invalid_prefixlen_diff:
343 throw logic_error(code, "invalid prefixlen_diff");
344 case error_code::new_prefix_must_be_shorter:
345 throw logic_error(code, "new prefix must be shorter");
346 case error_code::new_prefix_must_be_longer:
347 throw logic_error(code, "new prefix must be longer");
348 case error_code::cannot_set_prefixlen_diff_and_new_prefix:
349 throw logic_error(code, "cannot set prefixlen_diff and new_prefix");
350 case error_code::not_contained_network:
351 throw logic_error(code, "network is not a subnet of other");
352 default:
353 throw error(code, "unknown error");
354 }
355#else
356 if (IPADDRESS_IS_CONST_EVALUATED(length)) {
357 const auto _err = int(code) / (int(code) - int(code)); // invalid input string
358 }
359#endif
360}
