475 /**
476 * Checks if the given IP address is contained within this network.
477 *
478 * This method determines whether the provided IP address is part of the network
479 * represented by this ip network object, based on the network address and netmask.
480 *
481 * @code{.cpp}
482 * constexpr auto result1 = ipv4_network::parse("192.0.2.0/28").contains(ipv4_address::parse("192.0.2.6"));
483 * constexpr auto result2 = ipv4_network::parse("192.0.2.0/28").contains(ipv4_address::parse("192.0.3.6"));
484 *
485 * std::cout << std::boolalpha << result1 << std::endl;
486 * std::cout << std::boolalpha << result2 << std::endl;
487 *
488 * // out:
489 * // true
490 * // false
491 * @endcode
492 * @param[in] address The IP address to check.
493 * @return `true` if the address is part of the network, `false` otherwise.
494 */
495 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool contains(const ip_address_type& address) const IPADDRESS_NOEXCEPT {
496 const auto& ip_bytes = address.bytes();
497 const auto& address_bytes = network_address().bytes();
498 const auto& netmask_bytes = netmask().bytes();
499
500 for (size_t i = 0; i < ip_address_type::base_size; ++i) {
501 if ((ip_bytes[i] & netmask_bytes[i]) != address_bytes[i]) {
502 return false;
503 }
504 }
505 return true;
506 }

965 * // 192.0.2.128/26

967 * @endcode

970 * @param[in] new_prefixlen An optional new prefix length for the subnets. If not specified, the prefix length is determined by adding prefixlen_diff to the current prefix length.
971 * @return A `subnets_sequence` object representing the sequence of subnets, or an empty sequence if an error occurs.
972 * @remark `subnets_sequence` uses lazy evaluation to iterate over the subnets.

974 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE subnets_sequence<ip_network_base<Base>> subnets(error_code& code, size_t prefixlen_diff = 1, optional<size_t> new_prefixlen = nullptr) const IPADDRESS_NOEXCEPT {
975 auto address = Base::remove_scope_id(network_address());
976
977 if (prefixlen() == ip_address_type::base_max_prefixlen) {
978 return subnets_sequence<ip_network_base<Base>>(address, prefixlen());
979 }
980
981 if (new_prefixlen) {
982 if (new_prefixlen.value() < prefixlen()) {
983 code = error_code::NEW_PREFIX_MUST_BE_LONGER;
984 return subnets_sequence<ip_network_base<Base>>();
985 }
986 if (prefixlen_diff != 1) {
987 code = error_code::CANNOT_SET_PREFIXLEN_DIFF_AND_NEW_PREFIX;
988 return subnets_sequence<ip_network_base<Base>>();
989 }
990 prefixlen_diff = new_prefixlen.value() - prefixlen();
991 }
992
993 auto new_prefix = prefixlen() + prefixlen_diff;
994
995 if (new_prefix > ip_address_type::base_max_prefixlen) {
996 code = error_code::INVALID_PREFIXLEN_DIFF;
997 return subnets_sequence<ip_network_base<Base>>();
998 }
999
1000 return subnets_sequence<ip_network_base<Base>>(address, broadcast_address(), hostmask(), prefixlen_diff, new_prefix);
1001 }

1038 return ip_network_base<Base>();

1042 const auto address = ip_address_type::from_uint(network_address().to_uint() & (netmask().to_uint() << prefixlen_diff));
1043 return ip_network_base<Base>::from_address(address, new_prefix);
