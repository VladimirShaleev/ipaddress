466 /**
467 * Checks if the given IP address is contained within this network.
468 *
469 * This method determines whether the provided IP address is part of the network
470 * represented by this ip network object, based on the network address and netmask.
471 *
472 * @code{.cpp}
473 * constexpr auto result1 = ipv4_network::parse("192.0.2.0/28").contains(ipv4_address::parse("192.0.2.6"));
474 * constexpr auto result2 = ipv4_network::parse("192.0.2.0/28").contains(ipv4_address::parse("192.0.3.6"));
475 *
476 * std::cout << std::boolalpha << result1 << std::endl;
477 * std::cout << std::boolalpha << result2 << std::endl;
478 *
479 * // out:
480 * // true
481 * // false
482 * @endcode
483 * @param[in] address The IP address to check.
484 * @return `true` if the address is part of the network, `false` otherwise.
485 */
486 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool contains(const ip_address_type& address) const IPADDRESS_NOEXCEPT {
487 const auto& ip_bytes = address.bytes();
488 const auto& address_bytes = network_address().bytes();
489 const auto& netmask_bytes = netmask().bytes();
490
491 for (size_t i = 0; i < ip_address_type::base_size; ++i) {
492 if ((ip_bytes[i] & netmask_bytes[i]) != address_bytes[i]) {
493 return false;
494 }
495 }
496 return true;
497 }
