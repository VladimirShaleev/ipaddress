33template <typename T>
34struct ip_any_parser {
35 template <typename Str>
36 IPADDRESS_NODISCARD_WHEN_NO_EXCEPTIONS static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE T parse(const Str& address) IPADDRESS_NOEXCEPT_WHEN_NO_EXCEPTIONS {
37 auto code = error_code::NO_ERROR;
38 const auto ipv4 = ipv4_address::parse(address, code);
39 if (code == error_code::NO_ERROR) {
40 return T(ipv4);
41 }
42 return T(ipv6_address::parse(address));
43 }
44
45 template <typename Str>
46 static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE T parse(const Str& address, error_code& code) IPADDRESS_NOEXCEPT {
47 code = error_code::NO_ERROR;
48 const auto ipv4 = ipv4_address::parse(address, code);
49 if (code == error_code::NO_ERROR) {
50 return T(ipv4);
51 }
52 const auto ipv6 = ipv6_address::parse(address, code);
53 if (code == error_code::NO_ERROR) {
54 return T(ipv6);
55 }
56 return T();
57 }
58};

62/**
63 * A class that represents an IP address, supporting both IPv4 and IPv6 formats.
64 *
65 * The ip_address class is a union-type structure that encapsulates both ipv4_address
66 * and ipv6_address, allowing for the representation and manipulation of both IPv4
67 * and IPv6 addresses. It provides a set of functions and type aliases that facilitate
68 * the handling of IP addresses in a network context, making it a versatile tool for
69 * developers working with IP-based communications.
70 *
71 * The class ensures that the size of its instances will always be large enough to
72 * store an IPv6 address, which is the larger of the two address types. This design
73 * guarantees that an ip_address object can store any valid IP address, regardless
74 * of its version, without the need for separate storage mechanisms.

480 /**
481 * Static method template to parse an IP address from a fixed string at compile time.
482 *
483 * @tparam FixedString A non-type template parameter that holds the string representation of the IP address.
484 * @return An instance of ip address parsed from the fixed string.
485 * @note This method is only available when non-type template parameters for strings are supported.
486 * @remark If parsing fails, an error will be raised at compile time.
487 */
488 template <fixed_string FixedString>
489 IPADDRESS_NODISCARD static IPADDRESS_CONSTEVAL IPADDRESS_FORCE_INLINE ip_address parse() IPADDRESS_NOEXCEPT {
490 constexpr auto str = FixedString;
491 auto code = error_code::NO_ERROR;
492
493 char ip[str.size() + 1]{};
494 for (size_t i = 0; i < str.size(); ++i) {
495 ip[i] = str[i];
496 }
497
498 const auto ipv4 = ipv4_address::parse(ip, code);
499 if (code == error_code::NO_ERROR) {
500 return ip_address(ipv4);
501 }
502
503 return ip_address(ipv6_address::parse<FixedString>());
504 }
