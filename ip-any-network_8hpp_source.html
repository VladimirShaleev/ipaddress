35template <typename T>
36struct net_any_parser {
37 template <typename Str>
38 IPADDRESS_NODISCARD_WHEN_NO_EXCEPTIONS static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE T parse(const Str& address, bool strict) IPADDRESS_NOEXCEPT_WHEN_NO_EXCEPTIONS {
39 auto code = error_code::NO_ERROR;
40 const auto net4 = ipv4_network::parse(address, code, strict);
41 if (code == error_code::NO_ERROR) {
42 return T(net4);
43 }
44 return T(ipv6_network::parse(address, strict));
45 }
46
47 template <typename Str>
48 static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE T parse(const Str& address, error_code& code, bool strict) IPADDRESS_NOEXCEPT {
49 code = error_code::NO_ERROR;
50 const auto net4 = ipv4_network::parse(address, code, strict);
51 if (code == error_code::NO_ERROR) {
52 return T(net4);
53 }
54 const auto net6 = ipv6_network::parse(address, code, strict);
55 if (code == error_code::NO_ERROR) {
56 return T(net6);
57 }
58 return T();
59 }

555 * @throw logic_error Raise if network is not completely contained in this network.
556 * @remark `exclude_network_sequence` uses lazy evaluation to iterate networks.
557 */
558 IPADDRESS_NODISCARD_WHEN_NO_EXCEPTIONS IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE exclude_network_sequence<ip_network> address_exclude(const ip_network& other) const IPADDRESS_NOEXCEPT_WHEN_NO_EXCEPTIONS {
559 error_code code = error_code::NO_ERROR;
560 const auto result = address_exclude(other, code);
561 if (code != error_code::NO_ERROR) {
562 if (IPADDRESS_IS_CONST_EVALUATED(code)) {
563 raise_error(code, 0, "", 0);
564 }
565 #ifndef IPADDRESS_NO_EXCEPTIONS
566 raise_error(code, 0, "", 0);
567 #endif
568 }
569 return result;
570 }

572 /**
573 * Computes the network definitions resulting from removing the given network from this one, with error handling.
574 *
575 * @code{.cpp}
576 * constexpr auto a = ip_network::parse("192.0.2.0/28");
577 * constexpr auto b = ip_network::parse("192.0.2.1/32");
578 *
579 * auto err = error_code::NO_ERROR;
580 * auto exclude_sequence = a.address_exclude(b, err);
581 *
582 * if (err == error_code::NO_ERROR) {
583 * for (const auto& net : exclude_sequence) {
584 * std::cout << net << std::endl;
585 * }
586 * }
587 *
588 * // out:
589 * // 192.0.2.8/29
590 * // 192.0.2.4/30
591 * // 192.0.2.2/31
592 * // 192.0.2.0/32
593 * @endcode
594 * @param[in] other The other network to exclude from this network.
595 * @param[out] code An error_code object that will be set if an error occurs during the operation.
596 * @return An `exclude_network_sequence` object representing the remaining address ranges, or an empty sequence if an error occurs.
597 * @remark `exclude_network_sequence` uses lazy evaluation to iterate networks.
598 */
599 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE exclude_network_sequence<ip_network> address_exclude(const ip_network& other, error_code& code) const IPADDRESS_NOEXCEPT {
600 code = error_code::NO_ERROR;
601 if (_version != other._version) {
602 code = error_code::INVALID_VERSION;
603 return exclude_network_sequence<ip_network>();
604 }
605
606 auto addr1 = network_address(); addr1.set_scope_id("");
607 auto addr2 = other.network_address(); addr2.set_scope_id("");
608 auto lhs = ip_network::from_address(addr1, prefixlen());
609 auto rhs = ip_network::from_address(addr2, other.prefixlen());
610
611 if (!rhs.subnet_of(lhs)) {
612 code = error_code::NOT_CONTAINED_NETWORK;
613 return exclude_network_sequence<ip_network>();
614 }
615
616 if (lhs == rhs) {
617 return exclude_network_sequence<ip_network>();
618 }
619
620 return exclude_network_sequence<ip_network>(lhs, rhs);
621 }
