35template <typename T>
36struct net_any_parser {

40 const auto net4 = ipv4_network::parse(address, code, strict);

42 return T(net4);
43 }

47 template <typename Str>

49 code = error_code::no_error;

51 if (code == error_code::no_error) {
52 return T(net4);
53 }
54 const auto net6 = ipv6_network::parse(address, code, strict);
55 if (code == error_code::no_error) {
56 return T(net6);

58 return T();
59 }

553 * new prefix of the subnets; it must be larger than our prefix. One and only one of \a prefixlen_diff

576 * @remark `subnets_any_sequence` uses lazy evaluation to iterate over the subnets.

581 return subnets_any_sequence<ip_network>(sequence.begin(), sequence.end());
582 } else {

584 return subnets_any_sequence<ip_network>(sequence.begin(), sequence.end());

591 * @code{.cpp}
592 * constexpr auto a = ip_network::parse("192.0.2.0/28");

602 * // 192.0.2.4/30
603 * // 192.0.2.2/31
604 * // 192.0.2.0/32
605 * @endcode
606 * @param[in] other The other network to exclude from this network.
607 * @return An `exclude_network_sequence` object representing the remaining address ranges.
608 * @throw logic_error Raise if network is not completely contained in this network.
609 * @remark `exclude_network_sequence` uses lazy evaluation to iterate networks.
610 */
611 IPADDRESS_NODISCARD_WHEN_NO_EXCEPTIONS IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE exclude_network_sequence<ip_network> address_exclude(const ip_network& other) const IPADDRESS_NOEXCEPT_WHEN_NO_EXCEPTIONS {

620 /**
621 * Computes the network definitions resulting from removing the given network from this one, with error handling.
622 *
623 * @code{.cpp}
624 * constexpr auto a = ip_network::parse("192.0.2.0/28");
625 * constexpr auto b = ip_network::parse("192.0.2.1/32");
626 *
627 * auto err = error_code::no_error;
628 * auto exclude_sequence = a.address_exclude(b, err);
629 *
630 * if (err == error_code::no_error) {
631 * for (const auto& net : exclude_sequence) {
632 * std::cout << net << std::endl;
633 * }
634 * }
635 *
636 * // out:
637 * // 192.0.2.8/29
638 * // 192.0.2.4/30
639 * // 192.0.2.2/31
640 * // 192.0.2.0/32
641 * @endcode
642 * @param[in] other The other network to exclude from this network.
643 * @param[out] code An error_code object that will be set if an error occurs during the operation.
644 * @return An `exclude_network_sequence` object representing the remaining address ranges, or an empty sequence if an error occurs.
645 * @remark `exclude_network_sequence` uses lazy evaluation to iterate networks.
646 */
647 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE exclude_network_sequence<ip_network> address_exclude(const ip_network& other, error_code& code) const IPADDRESS_NOEXCEPT {
648 code = error_code::no_error;
649 if (_version != other._version) {
650 code = error_code::invalid_version;
651 return exclude_network_sequence<ip_network>();
652 }
653
654 auto addr1 = network_address(); addr1.set_scope_id("");
655 auto addr2 = other.network_address(); addr2.set_scope_id("");
656 auto lhs = ip_network::from_address(addr1, prefixlen());
657 auto rhs = ip_network::from_address(addr2, other.prefixlen());
658
659 if (!rhs.subnet_of(lhs)) {
660 code = error_code::not_contained_network;
661 return exclude_network_sequence<ip_network>();
662 }
663
664 if (lhs == rhs) {
665 return exclude_network_sequence<ip_network>();
666 }
667
668 return exclude_network_sequence<ip_network>(lhs, rhs);
669 }
