61 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE ip_address_base<Ext> ip_from_string(Iter begin, Iter end, error_code& code, int& index) IPADDRESS_NOEXCEPT {
62 if (begin == end) {
63 code = error_code::EMPTY_ADDRESS;
64 return {};
65 }
66
67 base_type octets = {};
68 char first_symbol = '\0';
69 int digits = 0;
70 int octet = 0;
71
72 index = 0;
73 code = error_code::NO_ERROR;
74
75 for (auto it = begin; it != end; ++it) {
76 auto c = char(*it);
77 if (index >= 4) {
78 code = error_code::EXPECTED_4_OCTETS;
79 return {};
80 }
81 if (c >= '0' && c <= '9') {
82 if (digits > 0 && first_symbol == '0') {
83 code = error_code::LEADING_0_ARE_NOT_PERMITTED;
84 return {};
85 } else if (digits == 0) {
86 first_symbol = c;
87 }
88 auto d = c - '0';
89 octet = octet * 10 + d;
90 ++digits;
91 if (digits > 3) {
92 code = error_code::OCTET_MORE_3_CHARACTERS;
93 return {};
94 }
95 } else if (c == '.' && digits > 0) {
96 if (octet > 255) {
97 code = error_code::OCTET_EXCEEDED_255;
98 return {};
99 }
100 octets[index++] = uint8_t(octet & 0xFF);
101 digits = 0;
102 octet = 0;
103 } else if (c == '.') {
104 code = error_code::EMPTY_OCTET;
105 return {};
106 } else {
107 code = error_code::OCTET_HAS_INVALID_SYMBOL;
108 return {};
109 }
110 }
111 if (index != 3) {
112 code = error_code::EXPECTED_4_OCTETS;
113 return {};
114 }
115 if (digits == 0) {
116 code = error_code::EMPTY_OCTET;
117 return {};
118 }
119 if (octet > 255) {
120 code = error_code::OCTET_EXCEEDED_255;
121 return {};
122 }
123 octets[index] = uint8_t(octet & 0xFF);
124 return ip_address_base<Ext>(octets);
125 }

127 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE ip_address_base<Ext> ip_from_uint32(uint_type ip) IPADDRESS_NOEXCEPT {
128 ip = is_little_endian() ? swap_bytes(ip) : ip;
129 return ip_address_base<Ext>({
130 uint8_t(ip & 0xFF),
131 uint8_t((ip >> 8) & 0xFF),

136 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE ip_address_base<Ext> ip_from_prefix(size_t prefixlen) {
137 return prefixlen != 0
138 ? ip_address_base<Ext>::ip_from_uint32(base_all_ones ^ (base_all_ones >> (prefixlen - 1) >> 1))
139 : ip_address_base<Ext>::ip_from_uint32(0);

142 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint_type ip_to_uint32(const base_type& bytes) IPADDRESS_NOEXCEPT {
143 const auto ip =
144 (uint32_t(bytes[3]) << 24) |
145 (uint32_t(bytes[2]) << 16) |
146 (uint32_t(bytes[1]) << 8) |
147 (uint32_t(bytes[0]));
148 return is_little_endian() ? swap_bytes(ip) : ip;
149 }

151 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t ip_to_chars(const base_type& bytes, format fmt, char (&result)[base_max_string_len + 1]) IPADDRESS_NOEXCEPT {
152 size_t offset = 0;
153 char buffer[4] {};
154 for (size_t b = 0; b < 4; ++b) {
155 size_t length = byte_to_string(bytes[b], buffer);
156 for (size_t i = 0; i < length; ++i) {
157 result[offset++] = buffer[i];
158 }
159 if (b < 3) {
160 result[offset++] = '.';
161 }
162 }
163 result[offset] = '\0';
164 return offset;

172 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE std::tuple<ip_address_base<Ext>, size_t> parse_netmask(Iter begin, Iter end, error_code& code, int& index) IPADDRESS_NOEXCEPT {
173 size_t prefixlen = 0;
174 auto is_value = true;
175 auto has_prefixlen = false;
176 for (auto it = begin; it != end; ++it) {
177 has_prefixlen = true;
178 const auto c = char(*it);
179 if (c >= '0' && c <= '9') {
180 prefixlen = prefixlen * 10 + (c - '0');
181 } else {
182 is_value = false;
183 break;
184 }
185 }
186 if (is_value) {
187 if (prefixlen > base_max_prefixlen) {
188 code = error_code::INVALID_NETMASK;
189 return std::make_tuple(ip_address_base<Ext>(), 0);
190 }
191 } else {
192 auto ip = ip_to_uint32(ip_from_string(begin, end, code, index).bytes());
193 if (code != error_code::NO_ERROR) {
194 code = error_code::INVALID_NETMASK;
195 return std::make_tuple(ip_address_base<Ext>(), 0);
196 }
197
198 prefixlen = prefix_from_ip_uint32(ip, code);
199 if (code != error_code::NO_ERROR) {
200 ip = ip ^ base_all_ones;
201 code = error_code::NO_ERROR;
202 prefixlen = prefix_from_ip_uint32(ip, code);
203 if (code != error_code::NO_ERROR) {
204 return std::make_tuple(ip_address_base<Ext>(), 0);
205 }
206 }
207 }
208 prefixlen = has_prefixlen ? prefixlen : base_max_prefixlen;
209 auto netmask = ip_from_prefix(prefixlen);
210 return std::make_tuple(netmask, prefixlen);
211 }

214 auto pack_address = address.to_uint();

228 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t prefix_from_ip_uint32(uint_type ip, error_code& code) IPADDRESS_NOEXCEPT {
229 auto trailing_zeroes = count_righthand_zero_bits(ip, base_max_prefixlen);
230 auto prefixlen = base_max_prefixlen - trailing_zeroes;
231 auto leading_ones = trailing_zeroes != 32 ? (ip >> trailing_zeroes) : 0;
232 auto all_ones = (uint_type(1) << (prefixlen - 1) << 1) - uint_type(1);
233 if (leading_ones != all_ones) {
234 code = error_code::NETMASK_PATTERN_MIXES_ZEROES_AND_ONES;
235 return 0;
236 }
237 return prefixlen;
238 }

240 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t count_righthand_zero_bits(uint32_t number, uint32_t bits) IPADDRESS_NOEXCEPT {
241 if (number == 0) {
242 return bits;
243 } else {
244 number = (~number & (number - 1));
245 size_t count = 0;
246 while (number != 0) {
247 count += number & 0x1;
248 number >>= 1;
249 }
250 if (bits < count) {
251 return bits;
252 } else {
253 return count;
254 }
255 }
256 }

258 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE size_t byte_to_string(uint8_t byte, char (&str)[4]) IPADDRESS_NOEXCEPT {
259 size_t length = 0;
260 char* p = str;
261 do {
262 *p++ = char((byte % 10) + '0');
263 byte /= 10;
264 ++length;
265 } while (byte != 0);
266
267 *p = '\0';
268
269 for (size_t i = 0, midle = length / 2; i < midle; ++i) {
270 auto tmp = str[i];
271 str[i] = *--p;
272 *p = tmp;
273 }
274
275 return length;
276 }
