29 static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void change(fixed_string<IPADDRESS_IPV6_SCOPE_MAX_LENGTH>& result, const Str& scope_id) IPADDRESS_NOEXCEPT_WHEN_NO_EXCEPTIONS {
30 error_code err = error_code::no_error;
31 change(result, scope_id, err);

39 template <typename Str>
40 static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void change(fixed_string<IPADDRESS_IPV6_SCOPE_MAX_LENGTH>& result, const Str& scope_id, error_code& code) IPADDRESS_NOEXCEPT {
41 code = error_code::no_error;
42 #if IPADDRESS_IPV6_SCOPE_MAX_LENGTH > 0
43 if (scope_id.size() > IPADDRESS_IPV6_SCOPE_MAX_LENGTH) {
44 code = error_code::scope_id_is_too_long;
45 return;
46 }
47 char scope[IPADDRESS_IPV6_SCOPE_MAX_LENGTH + 1] = {};
48 for (size_t i = 0; i < scope_id.size(); ++i) {
49 scope[i] = char(scope_id[i]);
50 }
51 result = make_fixed_string(scope);
52 #endif // IPADDRESS_IPV6_SCOPE_MAX_LENGTH
53 }

58/**
59 * Represents the scope identifier for an IPv6 address.
60 *
61 * The scope identifier is used to distinguish between different zones for the same address,
62 * such as link-local or site-local scopes. This class provides methods to set, retrieve,
63 * and compare scope identifiers in both string and numeric formats.
64 */
65class scope final {
66public:
67 /**
68 * Constructs a scope object with a given scope identifier.
69 *
70 * @param[in] scope_id A fixed_string representing the scope identifier.
71 */
72 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE scope(const fixed_string<IPADDRESS_IPV6_SCOPE_MAX_LENGTH>& scope_id) IPADDRESS_NOEXCEPT
73 :
74 _scope_id(scope_id) {
75 parse_value();
76 }
77
78 /**
79 * Retrieves the scope identifier as a string.
80 *
81 * @return A `std::string` representing the scope identifier.

83 IPADDRESS_NODISCARD IPADDRESS_FORCE_INLINE std::string get_string() const {
84 return std::string(_scope_id.begin(), _scope_id.end());

87 /**
88 * Retrieves the scope identifier as a numeric value.

90 * @return A uint32_t representing the numeric value of the scope identifier.
91 */
92 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint32_t get_uint32() const IPADDRESS_NOEXCEPT {
93 return _scope_id_value;
94 }
95
96 /**
97 * Checks if the scope identifier has a string representation.

99 * @return `true` if a string representation exists, `false` otherwise.

101 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool has_string() const IPADDRESS_NOEXCEPT {
102 return !_scope_id.empty();

104
105 /**
106 * Checks if the scope identifier has a numeric representation.

110 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE bool has_uint32() const IPADDRESS_NOEXCEPT {

215#endif // !IPADDRESS_HAS_SPACESHIP_OPERATOR

217private:
218 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void parse_value() IPADDRESS_NOEXCEPT {
219 if (!_scope_id.empty()) {
220 _has_value = true;
221 uint32_t value = 0;
222 for (auto c : _scope_id) {
223 if (c >= '0' && c <= '9') {
224 value = value * 10 + (c - '0');
225 } else {
226 _has_value = false;
227 break;
228 }
229 }
230 _scope_id_value = _has_value ? value : 0;
231 }
232 }
233
234 fixed_string<IPADDRESS_IPV6_SCOPE_MAX_LENGTH> _scope_id;
235 uint32_t _scope_id_value = 0;
236 bool _has_value = false;

251 /**
252 * Retrieves the scope identifier of the IPv6 address.

255 * It is particularly relevant for link-local and site-local addresses.
256 *

260 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE scope get_scope_id() const IPADDRESS_NOEXCEPT {
261 return scope(
262 #if IPADDRESS_IPV6_SCOPE_MAX_LENGTH > 0
263 _data.scope_id
264 #else // IPADDRESS_IPV6_SCOPE_MAX_LENGTH <= 0
265 make_fixed_string("")
266 #endif // IPADDRESS_IPV6_SCOPE_MAX_LENGTH <= 0
267 );
268 }

270 /**
271 * Sets the scope identifier of the IPv6 address.
272 *
273 * This function sets the scope identifier using a character array. The length of the array
274 * should not exceed `IPADDRESS_IPV6_SCOPE_MAX_LENGTH + 1`.
275 *
276 * @tparam T The character type of the scope identifier.
277 * @tparam N The size of the scope identifier array.
278 * @param[in] scope_id The character array representing the scope identifier.
279 * @remark If scope is disabled in settings (`IPADDRESS_IPV6_SCOPE_MAX_LENGTH == 0`) then this call will have no effect.
280 */
281 template <typename T, size_t N>
282 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void set_scope_id(const T(&scope_id)[N]) IPADDRESS_NOEXCEPT {
283 internal::is_char_type<T>();
284 #if IPADDRESS_IPV6_SCOPE_MAX_LENGTH > 0
285 static_assert(N <= IPADDRESS_IPV6_SCOPE_MAX_LENGTH + 1, "scope id is too long");
286 char str[IPADDRESS_IPV6_SCOPE_MAX_LENGTH + 1] = {};
287 for (size_t i = 0; i < N; ++i) {
288 str[i] = scope_id[i];
289 }
290 _data.scope_id = make_fixed_string(str);
291 #endif // IPADDRESS_IPV6_SCOPE_MAX_LENGTH
292 }

580 ip_address_base<ipv6_address_base> result;
581 auto& bytes = result._data.bytes;
582 int shift = 0;
583 int inc = 8;
584 if (is_little_endian()) {
585 shift = 56;
586 inc = -8;
587 }
588 for (int i = 0, s = shift; i < 8; ++i, s += inc) {
589 bytes[i] = uint8_t(ip.upper() >> s);
590 bytes[i + 8] = uint8_t(ip.lower() >> s);
591 }
592 return result;
593 }

596 * Converts the IPv6 address to an unsigned integer.
597 *
598 * @return The unsigned integer representation of the IPv6 address.
599 * @remark Bytes in integer are presented in **host byte order**.

608 shift = 56;
609 inc = -8;
