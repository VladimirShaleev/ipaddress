183 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE void parse_value() IPADDRESS_NOEXCEPT {
184 if (!_scope_id.empty()) {
185 _has_value = true;
186 uint32_t value = 0;
187 for (auto c : _scope_id) {
188 if (c >= '0' && c <= '9') {
189 value = value * 10 + (c - '0');
190 } else {
191 _has_value = false;
192 break;
193 }
194 }
195 _scope_id_value = _has_value ? value : 0;
196 }

538 /**
539 * Creates an IPv6 address from an unsigned integer using a template parameter.
540 *
541 * @tparam Ip The unsigned integer representing the IPv6 address.
542 * @return An instance of ip address representing the IPv6 address.
543 */
544 IPADDRESS_NODISCARD static IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE ip_address_base<ipv6_address_base> from_uint(uint_type ip) IPADDRESS_NOEXCEPT {
545 ip_address_base<ipv6_address_base> result;
546 auto& bytes = result._data.bytes;
547 int shift = 0;
548 int inc = 8;
549 if (is_little_endian()) {
550 shift = 56;
551 inc = -8;
552 }
553 for (int i = 0, s = shift; i < 8; ++i, s += inc) {
554 bytes[i] = uint8_t(ip.upper() >> s);
555 bytes[i + 8] = uint8_t(ip.lower() >> s);
556 }
557 return result;
558 }

560 /**
561 * Converts the IPv6 address to an unsigned integer.
562 *
563 * @return The unsigned integer representation of the IPv6 address.
564 * @remark Bytes in integer are presented in **host byte order**.
565 */
566 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE uint_type to_uint() const IPADDRESS_NOEXCEPT {
567 const auto& bytes = _data.bytes;
568 uint64_t upper = 0;
569 uint64_t lower = 0;
570 int shift = 0;
571 int inc = 8;
572 if (is_little_endian()) {
573 shift = 56;
574 inc = -8;
575 }
576 for (int i = 0, s = shift; i < 8; ++i, s += inc) {
577 upper |= uint64_t(bytes[i]) << s;
578 lower |= uint64_t(bytes[i + 8]) << s;
579 }
580 return uint_type(upper, lower);
581 }
