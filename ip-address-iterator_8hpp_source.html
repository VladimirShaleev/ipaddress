414 * The `ip_address_iterator` class template is specialized for `ip_address_base<Base>` to provide iteration functionality over IP addresses.
415 * It supports all operations typical for a random access iterator, allowing both forward and backward traversal with constant time complexity
416 * for increment, decrement, and arithmetic operations. This specialization is crucial for enabling seamless iteration over IP addresses,

859 *
860 * The hosts_sequence class template is specialized for `ip_address_base<Base>` to provide a container-like interface for IP address sequences.
861 * It allows for iteration over the host addresses in a network or subnet, excluding the network and broadcast addresses when appropriate.
862 * This specialization is essential for network-related operations that require processing of individual host addresses within a given range.
863 *
864 * @tparam Base The base type from which the ip_address_base is derived, representing the underlying IP address type.
865 * @remark When iterating, obtaining addresses occurs through lazy calculations.
866 */

871 using size_type = size_t; /**< The type used for representing the size of the sequence. */
872 using difference_type = typename value_type::uint_type; /**< The type used for representing differences between iterators. */
873 using pointer = value_type*; /**< The pointer type for the value_type. */
874 using const_pointer = const value_type*; /**< The const pointer type for the value_type. */
875 using reference = value_type&; /**< The reference type for the value_type. */
876 using const_reference = const value_type&; /**< The const reference type for the value_type. */
877

879 using const_iterator = ip_address_iterator<value_type>; /**< The const iterator type for iterating over the sequence. */

882 using const_reverse_iterator = ip_reverse_iterator<const_iterator>; /**< The const reverse iterator type for iterating over the sequence in reverse. */

884 /**
885 * Constructs a hosts_sequence with specified network parameters.
886 *
887 * Initializes the sequence based on the provided network address, broadcast address, and prefix lengths.
888 * The sequence excludes the network and broadcast addresses when appropriate, according to the prefix length.
889 *
890 * @param[in] network_address The network address of the subnet.
891 * @param[in] broadcast_address The broadcast address of the subnet.
892 * @param[in] prefixlen The prefix length of the subnet.
893 * @param[in] max_prefixlen The maximum prefix length possible for the IP version.
894 */
895 IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE hosts_sequence(const_reference network_address, const_reference broadcast_address, size_t prefixlen, size_t max_prefixlen) IPADDRESS_NOEXCEPT {
896 if (prefixlen == max_prefixlen - 1) {
897 const auto begin = value_type::from_uint(network_address.to_uint());
898 const auto end = value_type::from_uint(broadcast_address.to_uint() + 1);

904 _begin = const_iterator(begin);
