436 IPADDRESS_NODISCARD IPADDRESS_CONSTEXPR IPADDRESS_FORCE_INLINE ip_network_iterator base() const IPADDRESS_NOEXCEPT {
437 auto result = *this;
438 const auto old = result._it._offset;
439 result._it._offset += _step;
440 if (result._it._offset < old) {
441 result._it._carry = 1 - result._it._carry;
442 }
443 result._it._current = ip_address_type::from_uint(result._it._offset);
444 result._current = value_type::from_address(*result._it, result._prefixlen);
445 return result;
446 }

575 if (_other.subnet_of(_s1)) {
576 if (_current != _s2) {
577 _current = _s2;

579 _s1 = subnets[0];
580 _s2 = subnets[1];
581 } else {
582 *this = ip_exclude_network_iterator();
583 }
584 } else if (_other.subnet_of(_s2)) {
585 if (_current != _s1) {
586 _current = _s1;
587 const auto subnets = _s2.subnets();
588 _s1 = subnets[0];
589 _s2 = subnets[1];
590 } else {
591 *this = ip_exclude_network_iterator();
592 }
593 } else {
594 #ifndef IPADDRESS_NO_EXCEPTIONS
595 throw std::out_of_range("index out of range");
596 #endif
597 }
598 } else {
599 if (_s1 == _other) {
600 if (_current != _s2) {
601 _current = _s2;
602 } else {
603 *this = ip_exclude_network_iterator();
604 }
605 } else if (_s2 == _other) {
606 if (_current != _s1) {
607 _current = _s1;
608 } else {
609 *this = ip_exclude_network_iterator();
610 }
611 } else {
612 #ifndef IPADDRESS_NO_EXCEPTIONS

614 #endif
